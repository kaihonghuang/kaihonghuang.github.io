Tutorial3 - Function

Script 和 Function的区别： 有无（），可带额外输入

Script:

求1 + 2 + 3 + .. + 100

s = 0;
for i = 1 : 100
    s =  s + i;
end
disp(s);

转为Function:

function mysum()
    s = 0;
    for i = 1 : 100
        s =  s + i;
    end
    disp(s);
end

求1到n的和

%n - input parameter/arguments(参数)
function mysum(n)
    s = 0;
    for i = 1 : n
        s =  s + i;
    end
    disp(s);
end

求a到b的和，需要两个参数

% a,a+1,...+b
function mysum(a,b)
    s = 0;
    for i = a : b
        s =  s + i;
    end
    disp(s);
end

看个报错

v1 = [6 7 8 9]
v2 = [1 3 5 7 9]
s1 = sum(v1)
s2 = sum(v2)
s1 + s2

s1 = mysum(1, 3) %报错

如何解决：

function result = mysum(a,b)
    s = 0;
    for i = a : b
        s =  s + i;
    end
    result = s; %return 放回值
end

现在就可以了： 

（4+5+6+7）+ （11+12+13+14+15）
S1 = mysum(4,7)
S2 = mysum(11,15)
S1 + s2

约数/factor

例子： 8: 1,2,4,8
思路： 用循环一个个检测

function result = count_factors(n)
    count = 0;
    for i = 1 : n
        if mod(n,i) == 0
            count = count + 1;
        end
    end
    result = count;
end

质数/prime number.1和本身能被整除

Result = 1:是， Result = 0:不是
Variable scope，放在不同的文件中，同样的result不会有影响 

function result = is_prime(n)
    count = count_factors(n)
    if count == 2
        result = 1;
    else
        result = 0;
    end
end

Tutorial4 - 动画效果

动态正弦曲线

x = -2*pi : 0.1 : 2*pi;
y = sin(x);
h = plot(x, y)

for i = 1 : 100
    x = x + 01;
    y = sin(x);
    set(h, 'xData', x, 'yData', y)
    drawnow;
end

压缩拉伸动态

theta = -10 * pi: 0.1 : 10 * pi
x = cos(theta);
y = sin(theta);
z = theta;

h = plot3(x, y, z)
axis([-1, 1, -1, 1, -40, 40])

while true
    for i = 1 : 100
        z = 0.98 * z;
        set(h, 'xData', x, 'yData', y, 'zData', z);
        drawnow;
    end

    for i = 1 : 100
        z = z / 0.98;
        set(h, 'xData', x, 'yData', y, 'zData', z);
        drawnow;
    end
end

闹钟动态

t = 0 : pi/50 : 2 * pi;
x = cos(t);
y = sin(t);
plot(x, y);
axis equal;

hold on;

linex = [0, 1];
liney = [0, 0];
h = plot(linex, liney);

theta = 0;
for i = 1 : 100
    theta = theta + 0.1;
    linex(2) = cos(theta);
    liney(2) = sin(theta);
    set(h, 'xData', linex, 'yData', liney);
    drawnow;
end

Tutorial5 - 音乐制作

小星星

function y = gen_wave(tone, rythm)
    Fs = 8192;
    freqs = [524, 587, 659, 698, 783, 880, 988];
    x = linspace(0, 2 * pi * rythm, floor(Fs * rythm));
    y = sin(freqs(tone) * x) .* (1 - x / (rythm * 2 * pi));
end

Fs = 8192;

y1 = gen_wave(1, 0.5);
y2 = gen_wave(1, 0.5);
y3 = gen_wave(5, 0.5);
y4 = gen_wave(5, 0.5);
y5 = gen_wave(6, 0.5);
y6 = gen_wave(6, 0.5);
y7 = gen_wave(5, 1);
y = [y1, y2, y3, y4, y5, y6, y7];

y1 = gen_wave(4, 0.5);
y2 = gen_wave(4, 0.5);
y3 = gen_wave(3, 0.5);
y4 = gen_wave(3, 0.5);
y5 = gen_wave(2, 0.5);
y6 = gen_wave(2, 0.5);
y7 = gen_wave(1, 1);
y = [y, y1, y2, y3, y4, y5, y6, y7];

sound(y, Fs);
